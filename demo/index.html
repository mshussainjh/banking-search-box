Usage

Drop in any HTML page or framework — no build step required.

Code 
<!-- 1. Import the component -->
<script type="module" src="./banking-search-box.js"></script>

<!-- 2. Use it anywhere -->
<smart-search
  placeholder="Search accounts…"
  show-filters="true"
  filters='[{"id":"account","label":"Accounts","emoji":"🏦"}]'
  highlight="true"
  debounce="200"
></smart-search>

// 3. Listen for events
const search = document.querySelector('smart-search');

search.addEventListener('ss-search', async ({ detail }) => {
  const results = await fetchResults(detail.query);
  search.setResults(results);  // inject results back
});

search.addEventListener('ss-select', ({ detail }) => {
  console.log('Selected:', detail.item);
});
